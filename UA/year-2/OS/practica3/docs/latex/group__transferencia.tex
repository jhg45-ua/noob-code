Transfiere datos de un descriptor de origen a uno de destino. 

Lee todos los datos disponibles de {\ttfamily fd\+\_\+src} y los escribe en {\ttfamily fd\+\_\+dst} usando el buffer proporcionado. Funciona con archivos, pipes o sockets.


\begin{DoxyParams}[1]{Parámetros}
\mbox{\texttt{in}}  & {\em fd\+\_\+src} & Descriptor de origen (archivo, pipe, socket) \\
\hline
\mbox{\texttt{in}}  & {\em fd\+\_\+dst} & Descriptor de destino (archivo, pipe, socket) \\
\hline
\mbox{\texttt{in,out}}  & {\em buffer} & Buffer temporal para la transferencia (proporcionado por el usuario) \\
\hline
\mbox{\texttt{in}}  & {\em buffer\+\_\+size} & Tamaño del buffer en bytes\\
\hline
\end{DoxyParams}
\begin{DoxyReturn}{Devuelve}
Número total de bytes transferidos exitosamente 

-\/1 si hubo error de lectura o escritura
\end{DoxyReturn}
\begin{DoxyPrecond}{Precondición}
fd\+\_\+src debe estar abierto para lectura 

fd\+\_\+dst debe estar abierto para escritura

\end{DoxyPrecond}
\begin{DoxyNote}{Nota}

\end{DoxyNote}
